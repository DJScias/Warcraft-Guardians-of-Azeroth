# BUTCHERING EVENT
# Delayed event
narrative_event = {
	id = WCFW.62
	title = EVTTITLE_WCFW_62
	
	immediate = {
		d_frostwolf = {
			holder_scope = {
				save_event_target_as = target_frostwolf_leader
			}
		}
		d_stormreaver = {
			holder_scope = {
				save_event_target_as = target_frostwolf_butcher
				
				# Create the new character under Stormreaver
				create_character = {
					dynasty = actually_culture
					culture = stormreaver religion = orcish_fel
					trait=tough_soldier
					trait=creature_orc
					trait=class_warrior_5
					random_traits = yes
					age = 22
				}
				new_character = {
					save_event_target_as = target_new_orc
				}
			}
		}
	}
	
	desc = {
		trigger = { character = event_target:target_frostwolf_leader }
		text = EVTDESC_WCFW_62 # You are outside watching the campfire when you suddenly hear loud screams...
	}
	desc = {
		trigger = { NOT = { character = event_target:target_frostwolf_leader } }
		text = EVTDESC_WCFW_62_horde # Word arrives that the Frostwolf clan led by [target_frostwolf_leader.GetFirstName] has been completely decimated...
	}
	
	picture = GFX_evt_orcish_horde
	border = GFX_event_narrative_frame_war
	
	is_triggered_only = yes
	hide_from = yes
	
	option = {
		trigger = { character = event_target:target_frostwolf_leader }
		name = EVTOPT_WCFW_62 # May the spirits curse you forever, [From.GetFirstName]!
	}
	
	option = {
		trigger = { NOT = { character = event_target:target_frostwolf_leader } }
		name = INTERESTING
	}
	
	after = {
		# Create landless title for new orc (so he can be landless)
		create_title = {
			tier = DUKE
			landless = yes
			holder = event_target:target_new_orc
			culture = event_target:target_frostwolf_butcher
		}
		
		if = {
			limit = { character = event_target:target_frostwolf_leader }
			# Become adventurer orc if you're Frostwolf leader, they have a landless title so no game over
			set_player_character = event_target:target_new_orc
		}
		
		# Now butcher the whole frostwolf clan, abdicate to new orc + kill leader, killer unknown since done 'in secret'
		event_target:target_frostwolf_leader = {
			any_courtier = {
				death = { death_reason = death_murder_unknown killer = event_target:target_frostwolf_butcher }
			}
			
			# Destroy the Frostwolf tribe title and give the rest to the new orc
			destroy_landed_title = d_frostwolf
			abdicate_to = event_target:target_new_orc
			death = { death_reason = death_murder_unknown killer = event_target:target_frostwolf_butcher }
		}
		
		# Destroy the landless title so Dead Ravine becomes primary + make loyal vassal of Gul'dan/successor
		event_target:target_new_orc = {
			primary_title = {
				activate_title = { title = THIS status = no }
				destroy_landed_title = THIS
			}
			set_defacto_liege = event_target:target_frostwolf_butcher
			opinion = { modifier = opinion_loyal_servant who = event_target:target_frostwolf_butcher }
		}
		
		# Fire the event that speaks of the new inhabitants after three days (news has to travel, you know!)
		73 = {
			province_event = {
				id = WCFW.163
				days = 3
			}
		}
	}
}
